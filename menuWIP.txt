#include <GL/glut.h>

void doNothing(int msg){}

void aToggle(int msg) {
	switch (msg) {
		case 1:
			axesEnable = 1;
			break;
		case 2:
			axesEnable = 0;
			break;
	}
	glutPostRedisplay();
}
void sToggle(int msg) {
	switch (msg) {
		case 1:
			signEnable = 1;
			break;
		case 2:
			signEnable = 0;
			break;
	}
	glutPostRedisplay();
}
void setVMode(int msg) {
	switch (msg) {
		case 1:
			VIEWMODE = 1;
			break;
		case 2:
			VIEWMODE = 2;
			break;
		case 3:
			VIEWMODE = 3;
			break;
	}
	glutPostRedisplay();
}

void initMenu(){
	// Menu entry that changes the camera view
	int VIEWMODE = glutCreateMenu(setVMode);
	glutAddMenuEntry("glPerspective", 1);
	glutAddMenuEntry("glOrtho", 2);
	glutAddMenuEntry("Custom", 3);
	// Menu entry that controls the rendering of the axis
	int axesEnable = glutCreateMenu(aToggle);
	glutAddMenuEntry("Axes On", 1);
	glutAddMenuEntry("Axes Off", 2);
	// Menu entry that controls the rendering of the sign
	int signEnable = glutCreateMenu(sToggle);
	glutAddMenuEntry("Sign On", 1);
	glutAddMenuEntry("Sign Off", 2);
	// Menu entry that controls the render mode
	int FILLMODE = glutCreateMenu(FILLMODE);
	glutAddMenuEntry("Wire Frame", GL_LINE_LOOP);
	glutAddMenuEntry("Filled Faces", GL_POLYGON);
	// Menu entry that resets the figure
	int reset = glutCreateMenu(reset);
	glutAddMenuEntry("Reset Figure", 1);
	// Menu entry that terminates the program
	int quit = glutCreateMenu(quit);
	glutAddMenuEntry("Exit", 0);

	// Creates the menu
	int menu = glutCreateMenu(doNothing);
	glutAddSubMenu("Viewing mode", VIEWMODE);
	glutAddSubMenu("Axes", axesEnable);
	glutAddSubMenu("House Fill Mode", FILLMODE);

	glutAttachMenu(GLUT_MIDDLE_BUTTON);
}
